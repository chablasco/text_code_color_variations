<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <title>CB Code Viewer</title>
  <script src="https://cdnjs.cloudflare.com/ajax/libs/p5.js/1.6.0/p5.min.js"></script>
  <style>
    body {
      margin: 0;
      overflow: hidden;
      background: black;
    }
    canvas {
      display: block;
    }
  </style>
</head>
<body>
<script>
let words = [];
let codeLines;
let fontSize = 16;
let scrollY = 0;

function preload() {
  codeLines = loadStrings('assets/code.txt');
}

function setup() {
  createCanvas(windowWidth, windowHeight);
  textFont('Courier', fontSize);
  textSize(fontSize);
  noStroke();
  parseCode();
}

function draw() {
  background(0);
  for (let w of words) {
    w.display();
  }
}

function parseCode() {
  let xStart = 20;
  let yStart = 30;
  let x = xStart;
  let y = yStart;

  for (let line of codeLines) {
    let tokens = splitWithWhitespace(line);
    for (let t of tokens) {
      let w = new CodeWord(t, x, y);
      words.push(w);
      x += textWidth(t);
    }
    x = xStart;
    y += fontSize + 8;
  }
}

function splitWithWhitespace(line) {
  let tokens = [];
  let word = '';
  for (let char of line) {
    if (char === ' ' || char === '\t') {
      if (word.length > 0) {
        tokens.push(word);
        word = '';
      }
      tokens.push(char);
    } else {
      word += char;
    }
  }
  if (word.length > 0) tokens.push(word);
  return tokens;
}

function mouseWheel(event) {
  scrollY -= event.delta;
  scrollY = constrain(scrollY, -10000, 100); // Adjust for content height
}

class CodeWord {
  constructor(word, x, y) {
    this.word = word;
    this.x = x;
    this.y = y;
  }

  display() {
    let drawX = this.x;
    let drawY = this.y + scrollY;
    let w = textWidth(this.word);
    let h = fontSize;

    let hovered = mouseX > drawX && mouseX < drawX + w &&
                  mouseY > drawY - h && mouseY < drawY;

    if (hovered && this.word.trim() !== '') {
      fill(random(255), random(255), random(255));
      rect(drawX - 2, drawY - h + 4, w + 4, h + 4);
      fill(random(255), random(255), random(255));
    } else {
      fill(255);
    }

    text(this.word, drawX, drawY);
  }
}

function windowResized() {
  resizeCanvas(windowWidth, windowHeight);
}
</script>
</body>
</html>
